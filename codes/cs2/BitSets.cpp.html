/**
 * Sets Bitwise Operation - Union, Intersection, Subtraction and Powerset
 * Sets.cpp
 * Purpose : Reads the sets data from txt file and creates subsequent
 * linked lists. Then displays available sets and takes user input to
 * perform basic set operations on those sets and saves the result in an output txt file.
 *
 * @source bitset.txt
 * @author Areeb Jamal
 * @version 1.0 08/09/2015
 */

#include <iostream>
#include <fstream>
#include <sstream>
#include <cstdlib>
#include <cmath>

using namespace std;

typedef struct Set set;

ifstream in;
ofstream out;

/* Set Declaration */
struct Set{
	char name[100];
	set *next;     // Each new set is next to previous one
	set *child;    // Elements of set are their children
}*start, *current, *nul, *Universal;

/////////////// Functions Declaration ////////////////

// Miscellaneous
void print(string);
bool ifDuplicate(set*);
bool isSubset(set*);
bool validSize(set*);
bool setsValid();

// Set Data getters
int getSize(set*);
string getSetName(string);
string getSetElements(string);
set* search(string);
string get(set*, int);

// String Modifiers
void stripAll(string&);
string stripString(string);

// BitWise Set Operations
int getBits(string);
string printBitSet(int, set* BitSet=Universal);
string powerset(string setA);

// Structure Linking
void linkElements(string);
////////////////////////////////////////////////////////

/* Main Program Control */
int main()
{
	in.open("bitset.txt");

	while(!in.is_open()){
		cout<<"Error opening file. File may not exist or is inaccessible.\nPress \'Enter\' to retry opening or \'Q\' to quit..."<<endl;
		if(cin.get()=='Q')
			return 0;
	}
	out.open("output.txt");

	stringstream buffer;
	buffer<<in.rdbuf();
	in.close();
	string a = buffer.str();
	linkElements(a);

	if(!setsValid())
		return 0;

	cout<<"Available sets :- ";
	current = start;
	while(current->next){
		cout<<current->name<<"|";
		current=current->next;
	}
	cout<<"\nPlease enter the sets to operate upon :\n\nA <U> B = A \u222A B\nA <I> B = A \u2229 B\nA <-> B = A - B\npowerset(A) = Powerset of set A\n"<<endl;

	string input;
	while(input!="Quit"){
		getline(cin, input);
		stripAll(input);
		string setA = input.substr(0, input.find("<"));
		string setB = input.substr(input.find(">")+1);

		if(input.find("powerset(")!=string::npos){
			setA = input.substr(input.find("(")+1);
			setA.erase(setA.length()-1);
			if(search(setA)){
				print(powerset(setA));
				continue;
			}
		}

		if(!search(setA)||!search(setB)){
			if(input=="Quit")
				break;
			else if(input.find("<U>")!=string::npos||input.find("<I>")!=string::npos||input.find("<->")!=string::npos)
				cout<<"Set(s) not found!"<<endl;
			else
				cout<<"Invalid Input"<<endl;
			continue;
		}

		if(input.find("<U>")!=string::npos)
			print(setA + " \u222A " + setB + " = " + printBitSet(getBits(setA)|getBits(setB)) + "\n");
		else if(input.find("<I>")!=string::npos)
			print(setA + " \u2229 " + setB + " = " + printBitSet(getBits(setA)&getBits(setB)) + "\n");
		else if(input.find("<->")!=string::npos)
			print(setA + " - " + setB + " = " + printBitSet(getBits(setA)&~getBits(setB)) + "\n");

	}

	out.close();
	cout<<"Program Completed"<<endl;
}

////////////  Function Definitions ////////////

/**
 * Unified Print Function
 * Prints a string to both console and file
 *
 * @param s The string to be printed
 */
void print(string s){
	cout<<s<<endl;
	out<<s<<endl;
}

/**
 * Set Size Checker
 * As sets need to have less than 32 elements ~ bits as int only supports
 * 32 bits, it checks if set @A has less than 32 elements and returns true, else false
 *
 * @param  A Pointer to the set which needs to be checked for size
 * @return boolean - True if there are less than 32 elements in set* @A, else false
 */
bool validSize(set* A){
	if(getSize(A)>32){
		cout<<"Sets cannot support more than 32 elements ~ bits! Program will terminate."<<endl;
		return false;
	}
	return true;
}

/**
 * Duplicate Element Checker
 * As sets need to be unique, checks for any duplicate element in a set
 * and prints it and Set name and returns true, else false
 *
 * @param  A Pointer to the set which needs to be checked for duplicates
 * @return boolean - True if there is a duplicate in set* @A, else false
 */
bool ifDuplicate(set* A){
	for(int i = 0; i < getSize(A); i++){
		for(int j = i+1; j < getSize(A); j++){
			if(get(A,i)==get(A,j)){
				cout<<"Error\nDuplicate element \'"<<get(A, i)<<"\' found in set "<<A->name<<endl;
				return true;
			}
		}
	}
	return false;
}

/**
 * Subset Checker
 * As a set needs to be a subset of Universal set, checks set @A
 * that no element lies outside Universal set and returns true, else false
 *
 * @param  A Pointer to the set which needs to be checked for being subset of Universal set
 * @return boolean - True if set* @A is subset of Universal set, else false
 */
bool isSubset(set* A){
	set* Universal = search("U");
	for(int i = 0; i < getSize(A); i++){
		bool notFound = true;
		for(int j = 0; j < getSize(Universal); j++){
			if(get(A, i)==get(Universal, j)){
				notFound = false;
				break;
			}
		}
		if(notFound){
			cout<<"Set "<<A->name<<" is not subset of Universal Set! Program will terminate."<<endl;
			return false;
		}
	}
	return true;
}

/**
 * Set Validity Checker
 * As all sets need to be unique, checks for any duplicate element in a set
 * for all sets by @ifDuplicate function and prints it.
 * As bitwise operations on int are limited to 32 bits, checks the size of
 * set by @validSize to ensure it is less than 32.
 * As all other sets need to be subsets of Universal set, checks all sets
 * using @isSubset function that no set lies outside Universal set.
 *
 * @return boolean - True if all the 3 conditions are satisfied by all sets, else false
 */
bool setsValid(){
	set *names = start;
	if(!search("U")){
		cout<<"Universal set not found in input file. Program will terminate"<<endl;
		return false;
	}
	Universal = search("U");
	while(names){
		if(ifDuplicate(names))
			return false;
		else if(!isSubset(names))
			return false;
		else if(!validSize(names))
			return false;
		else
			names=names->next;
	}
	return true;
}

/**
 * Set Size Getter
 * Calculates and returns the size, or number of elements in a set by incrementing
 * a counter till set value reaches null.
 *
 * @param  A Pointer to the set which needs to be checked for size
 * @return integer - The size of set provided
 */
int getSize(set* A){
	int i = 0;
	while(A){
		A=A->child;
		i++;
	}
	return i-1;
}

/**
 * Set Name Getter
 * Parses input string line from txt file to get and return set name
 * First, uses @stripAll function to remove any whitespaces, then finds
 * the index to 'SET' and '=' and returns the string bound by them
 *
 * @param  line The string containing set name
 * @return string - The parsed set name contained in input string
 */
string getSetName(string line){
	stripAll(line);
	size_t begin = line.find("SET")+3;
	size_t end = line.find("=")-line.find("SET")-3;
	return line.substr(begin, end);
}

/**
 * Set Elements Getter
 * Parses input string line from txt file to get and return set elements
 * First, uses @stripString to single out all excess whitespaces to one and
 * then return the sub-string bounded by '{' and '}'
 *
 * @param  line The string containing set elements
 * @return string - The parsed set elements contained in input string
 */
string getSetElements(string line){
	line = stripString(line);
	size_t begin = line.find("{")+1;
	size_t end = line.find("}")-line.find("{")-1;
	return stripString(line.substr(begin, end));
}

/**
 * Set Search
 * Iterates through -next- from starting Set and rerurns the Set
 * with name matching the parameter, if no matching set is found,
 * returns nul set
 *
 * @param  A string with which set names are to be matched
 * @return Set* - with name matching param, else nul
 */
set* search(string name){
	current = start;
	while(current){
		if(current->name==name)
			return current;
		else
			current=current->next;
	}
	return nul;
}

/**
 * Element Getter
 * Iterates and returns the nth element from input set
 * Iterates through param set* @A till *max* its size using
 * @getSize param @index times and returns tge reached element.
 *
 * @param  A Pointer to the set of which element needs to be returned
 * @param  index The value for index'th element to be returned
 * @return string - The name of index'th element in param set* @A
 */
string get(set* A, int index){
	int i = getSize(A);
	index++;
	if(index<=i){
		while(index--){
			A=A->child;
		}
		return A->name;
	}else{
		cout<<"Index Out of Bound"<<endl;
		return "";
	}
}

/**
 * String Strict Strip
 * First uses @stripString to remove all excess whitespaces and then
 * removes all occurrences of whitespaces in the param string reference
 *
 * @param  str String reference to be stripped
 */
void stripAll(string &str){
	str=stripString(str);
	for(int i = 0; i < str.length(); i++){
		if(str[i]==' '||str[i]=='\t'){
			str.erase(str.begin()+i, str.begin()+i+1);
		}
	}
	str[str.length()]='\0';
}

/**
 * String Strip
 * Generates and returns new string ignoring all double whitespaces and adding
 * any other characters. Resulting string has only single occurrences of whitespace
 * Then checks availability of any leading or trailing whitespaces and removes them
 *
 * @param  str String to be stripped to remove excess whitespaces
 * @return stripped string
 */
string stripString(string str){
	string stripped;
	for(int i = 0; i < str.length(); i++){
		if(str[i]=='\t')
			str[i]=' ';
		if(str[i]==' '&&str[i+1]==' ')
			continue;
		else
			stripped+=str[i];
	}
	if(stripped[stripped.length()-1]==' ')
		stripped.erase(stripped.length()-1);
	if(stripped[0]==' ')
		stripped.erase(0, 1);

	stripped[stripped.length()]='\0';
	return stripped;
}

/**
 * Bit Sets Getter
 * Searches elements of Universal set in @setA and returns corrensponding bit set in int
 * with each ith bit corresponding to ith element of Unoversal set being found in set or not.
 * 1 - if found or 0 - if not found.
 *
 * @param  setA String of Set name of which bitset corresponding to Universal Set is needed
 * @return int - bitset corresponding to availability of Universal Set elements in @setA
 */
int getBits(string setA){
	set* A = search(setA);
	int bits = 0;
	for(int i = 0; i < getSize(Universal); i++){
		for(int j = 0; j < getSize(A); j++){
			if(get(Universal, i)==get(A, j))
				bits |= 1<<i;
		}
	}
	return bits;
}

/**
 * Bit Set Printer
 * Returns a string to be printed with elements of Universal Set corresponding to respective elements
 * of input @ints and @BitSet
 * Default value of @BitSet is Universal Set
 *
 * Adds ith element of @BitSet in string if ith bit of @bits is 1
 *
 * @param  bits    int containing bits as information as to what to print from @BitSet
 * @param  BitSet  Pointer to set of which's elements are to be printed
 * @return toPrint String containing corresponding elements of @BitSet according to @bits
 */
string printBitSet(int bits, set* BitSet){
	bool ran = false;
	string toPrint = "{";
	for(int i = 0; i < getSize(BitSet); i++){
		if(bits&(1<<i)){
			ran = true;
			toPrint+=get(BitSet, i)+" ";
		}
	}
	if(ran)
		toPrint.erase(toPrint.length()-1);
	toPrint+="}";
	return toPrint;
}

/**
 * Powerset Generating Function
 * Iterates from 0 to 2^(size of @setA)-1 which givws us all iterants of powerset of A.
 * Then using those iterants, we get corresponding set elements of @setA using @printBitSet function
 *
 * Checks size of Set too and warns user if size is > 15 as it may consume time and memory to generate
 * powerset of that set.
 *
 * @param  setA String of set name of which powerset is to be generated
 * @return powerset of @setA
 */
string powerset(string setA){
	set* A = search(setA);
	string s = "Powerset of Set" + setA + " is\n";
	for(int i = 0; i < pow(2, getSize(A)); i++){
		s+=printBitSet(i, A)+"\n";
	}
	return s;
}

/**
 * Element Linking
 * Reads input string and creates set structures with elements linked to
 * each other.
 * Sets are linked to each other in next fashion and elements are in child
 * form.
 *
 * @param input string containing set and element information
 */
void linkElements(string input){
	stringstream lines(input);

	start = (set*) malloc(sizeof(set));
	set *top = start;
	for(string line; getline(lines, line);){
		if(!line.empty()){
			top->name[getSetName(line).copy(top->name, getSetName(line).length())] = '\0';
			line = getSetElements(line);
			current = top;
			top->next =(set*) malloc(sizeof(set));
			top = top->next;

			stringstream words(line);
			for(string word; getline(words, word, ' ');){
				current->child=(set*) malloc(sizeof(set));
				current->child->name[word.copy(current->child->name, word.length())] = '\0';
				current=current->child;

			}
		}
	}
	top->next=nul;
	current = start;
}
