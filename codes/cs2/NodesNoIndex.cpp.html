#include <iostream>
#include <cstdlib>

using namespace std;

typedef struct Node node;

static int size, rows, columns;

struct Node{
	int id;          //Exclusive ID set for each node for easy searching and pointing
	char name[20];
	node *up;
	node *down;
	node *left;
	node *right;
	node *next;      //Each node is first linearly linked for iterating through all elements.
}*start, *nul;

void setSize(int n){
	size = n;
}

int getSize(){
	return size;
}

void setRows(int n){
	rows = n;
}

int getRows(){
	return rows;
}

void setColumns(int n){
	columns = n;
}

int getColumns(){
	return columns;
}

void createNodes(node* curr=start){
	for(int i = 0; i < getSize(); i++){
		curr->next = (node*) malloc(sizeof(node));
		cout<<"Enter name for Node "<<i+1<<" :     ";
		cin.getline(curr->name, 20);
		curr->id=i+1;
		curr=curr->next;
	}
	nul = (node*) malloc(sizeof(node));
}

node* findNode(int id, node* curr=start){
	for(int i = 0; i < getSize(); i++){
		if(curr->id==id){
			return curr;
		}
		if(curr->next)
			curr=curr->next;
		else
			break;
	}
	return nul;
}

node* searchNode(string name, node *curr=start){
	for(int i = 0; i < getSize(); i++){
		if(curr->name==name){
			return curr;
		}
		if(curr->next)
			curr=curr->next;
		else
			break;
	}
	return nul;
}

node* getNode(){
	bool found = false;
	node *tmp;
	while(!found){
		string name;
		cout<<"Enter starting Node :    ";
		getline(cin, name);
		tmp = searchNode(name);
		if(tmp!=NULL){
			found = true;
			break;
		}
		cout<<"No such node found. Please try again..."<<endl;
	}
	return tmp;
}

int filterRight(int inp){
	
	if(!((inp-1)%getColumns()))
		return getSize()+1;
	else
		return inp;
}

int filterLeft(int inp){
	
	if(!(inp%getColumns()))
		return getSize()+1;
	else
		return inp;
}

void arrangeNodes(){
	for(int i = 1; i <=getSize(); i++){
		findNode(i)->up=findNode(i-getColumns());
		findNode(i)->down=findNode(i+getColumns());
		findNode(i)->right=findNode(filterRight(i+1));
		findNode(i)->left=findNode(filterLeft(i-1));
	}
}

int main()
{
	
	setRows(3);
	setColumns(3);
	setSize(getColumns()*getRows());
	start = (node*) malloc(sizeof(node));
	createNodes();
	arrangeNodes();
	node *tmp = getNode();
    
	cout<<"Enter moves"<<endl;
	while(tmp){
		char mov;
		cin>>mov;
		if(mov=='u')
			tmp=tmp->up;
		if(mov=='d')
			tmp=tmp->down;
		if(mov=='r')
			tmp=tmp->right;
		if(mov=='l')
			tmp=tmp->left;
			
		if(tmp&&tmp->id)
			cout<<tmp->name<<endl;
		else{
			cout<<"Null"<<endl;
			break;
		}
		
	}
	return 0;
}
